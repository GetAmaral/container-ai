       useEffect(() => {
         const el = document.getElementById("pricing");
         if (!el) return;
         const observer = new IntersectionObserver(
           ([entry]) => {
             if (entry.isIntersecting && !pricingTracked.current) {
               trackScrollToPrice();
               pricingTracked.current = true;
             }
           },
           { threshold: 0.3 }
         );
         observer.observe(el);
         return () => observer.disconnect();
       }, []);
